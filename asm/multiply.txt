; a = 5
imm1h 0x0
imm1l 0x5
mov r2 r1

; b = 3
imm1h 0x0
imm1l 0x3
mov r3 r1

; clear result and counter
mov r1 r0
mov r4 r0

label loop
    ; add a to res
    mov r1 r4
    add r2
    mov r4 r1

    ; decrement b
    imm1l 0xf
    imm1h 0xf
    add r3
    mov r3 r1

    ; loop if b is not zero
    jnz
    loop

halt
